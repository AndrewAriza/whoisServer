# Install CockroachDB
https://www.cockroachlabs.com/docs/stable/install-cockroachdb-mac.html

# Start CockroachDB
    cockroach start --certs-dir=certs --listen-addr=localhost

# Prepare
    make init

# Start
    make run

# Build
    make build
